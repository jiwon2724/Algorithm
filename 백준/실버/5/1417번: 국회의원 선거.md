fun main()  {
    val n = readln().toInt()
    var one = 0
    var count = 0
    val list = arrayListOf<Int>()
    repeat(n) { if (it == 0) one = readln().toInt() else list.add(readln().toInt()) }

    while (true) {
        if (list.all { it < one }) break
        val index = list.indexOf(list.maxOf { it })
        list[index]--
        one++
        count++
    }
    println(count)
}
