lateinit var array: Array<Array<Int>>

fun main() {
    val input = readln().split(" ").map { it.toInt() }
    val h = input[0]
    val w = input[1]
    var answer = 0
    var wall = 0
    array = Array(h) { Array(w) { 0 } }
    val block = readln().split(" ").map { it.toInt() }
    block.forEachIndexed { index, i ->
        for (j in h-1 downTo h-i) {
            array[j][index] = 1
        }
    }

    for (i in 0 until h) {
        var rain = 0
        for (j in 0 until w) {
            if (array[i][j] != 0) {
                wall++
                if (wall == 2) {
                    wall = 1
                    answer += rain
                    rain = 0
                }
            } else {
                if (wall == 1) {
                    rain++
                }
            }
        }
        wall = 0
    }
    println(answer)
}
