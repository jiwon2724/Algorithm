class Solution {
    fun solution(s: String): String {
        val temp = s.split(" ")
        var temp2 = IntArray(temp.size) { 0 }
        for(i in 0..temp.size-1) { temp2[i] = temp[i].toInt() }
        val answer = temp2.min().toString() + " " + temp2.max().toString()
        return answer
    }
}
